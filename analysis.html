lm_plot <- function(category) {
  temp <- data |>
    filter(EMP > 0) |>
    mutate(average_income = PAYANN / EMP * 1000) |>
    filter(
      NAICS2017_LABEL == category,
      ETH_GROUP_LABEL == "Total",
      SEX_LABEL %in% c("Male", "Female"),
      VET_GROUP_LABEL == "Total"
    )  
  return(
    ggplot(temp) +
      geom_point(aes(y = average_income, x = Population)) +
      geom_smooth(aes(y = average_income, x = Population), 
                  method = "lm") +
      ylab("Income / $") +
      ggtitle(strwrap(category,
                      width = 30)) +
      theme(
        plot.title = element_text(size = 10, vjust = 1)  # Customize size and alignment
      )
  )
}
